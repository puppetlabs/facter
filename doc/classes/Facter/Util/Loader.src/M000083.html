  def search_path
    result = []
    result += $LOAD_PATH.collect { |d| File.join(d, "facter") }
    if ENV.include?("FACTERLIB")
      result += ENV["FACTERLIB"].split(":")
    end

    # This allows others to register additional paths we should search.
    result += Facter.search_path

    result
  end
