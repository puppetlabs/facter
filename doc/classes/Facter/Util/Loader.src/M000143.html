  def load_all
    return if defined?(@loaded_all)

    load_env

    search_path.each do |dir|
      next unless FileTest.directory?(dir)

      Dir.entries(dir).sort.each do |file|
        path = File.join(dir, file)
        if File.directory?(path)
          load_dir(path)
        elsif file =~ /\.rb$/
          load_file(File.join(dir, file))
        end
      end
    end

    @loaded_all = true
  end
