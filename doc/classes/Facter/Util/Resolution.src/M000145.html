  def self.have_which
    if ! defined?(@have_which) or @have_which.nil?
      if Facter.value(:kernel) == 'windows'
        @have_which = false
      else
        %x{which which >/dev/null 2>&1}
        @have_which = ($? == 0)
      end
    end
    @have_which
  end
