  def self.get_vlans
    vlans = Array.new
    if self.get_vlan_config
      self.get_vlan_config.each_line do |line|
        if line =~ /^([0-9A-Za-z]+)\.([0-9]+) /
          vlans.insert(-1, $~[2]) if $~[2]
        end
      end
    end

    vlans.join(',')
  end
