def run_spec_files(files)
  files = Array(files)
  return if files.empty?
  opts = File.readlines('.rspec').collect { |l| l.chomp }.join(" ")
  begin
    run_spec("rspec --tty #{opts} #{files.join(' ')}")
  rescue => detail
    puts detail.backtrace
    warn "Failed to run #{files.join(', ')}: #{detail}"
  end
end
